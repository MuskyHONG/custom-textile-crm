# 定制家居织物品牌客服工具技术架构设计文档

## 1. 架构概述

### 1.1 设计原则
- **模块化设计**：将系统划分为独立的模块，便于开发、测试和维护
- **分层架构**：采用前后端分离的分层架构，提高系统的可扩展性和可维护性
- **RESTful API**：使用RESTful风格的API接口，便于系统集成和扩展
- **安全性**：实现基于JWT的认证机制，确保系统安全

## 7. 核心功能实现

### 7.1 价格计算功能

#### 7.1.1 基础价格计算
```typescript
// 基础价格 = 材质单价 × 面积
function calculateBasePrice(material: string, width: number, height: number): number {
  const materialPrices = {
    'cotton': 100,   // 元/平方米
    'linen': 150,    // 元/平方米
    'silk': 300,     // 元/平方米
    'wool': 200      // 元/平方米
  };
  
  const pricePerSquareMeter = materialPrices[material] || 100;
  const area = width * height / 10000; // 转换为平方米
  
  return pricePerSquareMeter * area;
}
```

#### 7.1.2 工艺费用计算
```typescript
// 工艺费用 = 基础价格 × 工艺系数
function calculateProcessCost(basePrice: number, process: string): number {
  const processFactors = {
    'embroidery': 0.3,    // 绣花工艺附加30%
    'printing': 0.15,      // 印花工艺附加15%
    'jacquard': 0.4,       // 提花工艺附加40%
    'plain': 0             // 平纹工艺无附加
  };
  
  const factor = processFactors[process] || 0;
  
  return basePrice * factor;
}
```

#### 7.1.3 折扣计算
```typescript
// 折扣 = （基础价格 + 工艺费用） × 折扣率
function calculateDiscount(basePrice: number, processCost: number, customerLevel: string): number {
  const discountRates = {
    'vip': 0.2,      // VIP客户20%折扣
    'regular': 0.1,  // 常规客户10%折扣
    'new': 0         // 新客户无折扣
  };
  
  const rate = discountRates[customerLevel] || 0;
  const subtotal = basePrice + processCost;
  
  return subtotal * rate;
}
```

#### 7.1.4 税费计算
```typescript
// 税费 = （基础价格 + 工艺费用 - 折扣） × 税率
function calculateTax(basePrice: number, processCost: number, discount: number, taxRate: number = 0.13): number {
  const taxableAmount = basePrice + processCost - discount;
  
  return taxableAmount * taxRate;
}
```

#### 7.1.5 最终报价汇总
```typescript
// 最终报价 = 基础价格 + 工艺费用 - 折扣 + 税费
function calculateFinalPrice(basePrice: number, processCost: number, discount: number, tax: number): number {
  return basePrice + processCost - discount + tax;
}
```

### 7.2 订单生成功能

```typescript
async function createOrder(projectId: string, customerId: string): Promise<Order> {
  // 获取项目信息
  const project = await Project.findById(projectId);
  if (!project) {
    throw new Error('Project not found');
  }
  
  // 生成订单编号
  const orderNumber = `ORD${Date.now()}`;
  
  // 创建订单
  const order = new Order({
    projectId,
    customerId,
    orderNumber,
    totalAmount: project.finalPrice,
    status: 'pending'
  });
  
  // 保存订单
  await order.save();
  
  // 更新项目状态
  project.status = 'ordered';
  await project.save();
  
  return order;
}
```

```typescript
function exportQuoteToPDF(quote: Quote): Buffer {
  // 使用PDF库生成报价单
  // 这里使用伪代码示例
  const pdf = new PDFDocument();
  
  // 添加报价单内容
  pdf.text('报价单', { align: 'center' });
  pdf.text(`客户名称: ${quote.customerName}`);
  pdf.text(`项目名称: ${quote.projectName}`);
  pdf.text(`基础价格: ¥${quote.basePrice.toFixed(2)}`);
  pdf.text(`工艺费用: ¥${quote.processCost.toFixed(2)}`);
  pdf.text(`折扣: ¥${quote.discount.toFixed(2)}`);
  pdf.text(`税费: ¥${quote.tax.toFixed(2)}`);
  pdf.text(`最终报价: ¥${quote.finalPrice.toFixed(2)}`);
  
  // 生成PDF缓冲区
  const buffer = await new Promise<Buffer>((resolve) => {
    const chunks: Buffer[] = [];
    pdf.on('data', (chunk) => chunks.push(chunk));
    pdf.on('end', () => resolve(Buffer.concat(chunks)));
    pdf.end();
  });
  
  return buffer;
}
```
